    ]
    % draw sites
    \foreach \i [remember=\i as \lasti, evaluate=\i as \filling using -0.1*\i+0.9] in {1,...,7}
        {
            \ifnum\i=1

                \node [bath={\filling}, label=right:$b_\i$] (b\i) [above=of b\lasti] {};
            \fi
        }

    % connect sites
    \foreach \i in {1,...,7}
        {
            \draw (impurity.east) to [out=0,in=180,looseness=0.3] (b\i.west);
        }
